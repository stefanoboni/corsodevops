pipeline {
  environment {
    registry = "stefanoboni/devops_api"
    registryCredential = 'dockerhub'
    dockerImage = ''
    dockerfile = 'devops/Dockerfile'
  }
  agent any
  stages {
    stage('Cloning Git') {
      steps {
          git branch: 'master',
                credentialsId: 'git',
                url: 'git@github.com:stefanoboni/devops.git'
      }
    }
    stage('Build from Dockerfile') {
       steps {
           script {
                dockerImage = docker.build("$registry:$BUILD_NUMBER", "-f ${dockerfile} .")
           }
       }
    }
    stage('Test') {
      steps {
        sh 'echo "Ciao"'
      }
    }
    stage('Deploy Image') {
      steps{
         script {
            docker.withRegistry( '', registryCredential ) {
            dockerImage.push()
          }
        }
      }
    }
    stage('Remove Unused docker image') {
      steps{
        sh "docker rmi $registry:$BUILD_NUMBER"
      }
    }
  }
}
